167void SevenSegmentDisplay<SEGMENTS_PORT, SEGMENT_FIRST_PIN, DIGITS_PORT, DIGITS_FIRST_PIN>::forceUpdate()
168{
169 (&DIGITS_PORT)->BSRR = dig0 | dig1 | dig2 | dig3;
170
171 switch(currentDisplayDigit)
172 {
173 case 0:{
174 uint16_t code = codeOfDigit(number / 1000) | codeOfDots(3);
175 (&SEGMENTS_PORT)->BSRR = code | (~code << 16);
176 (&DIGITS_PORT)->BRR = dig0;
177 currentDisplayDigit = 1;
178 break;
179 }
180 case 1:{
181 uint16_t code = codeOfDigit((number / 100) % 10) | codeOfDots(2);
182 (&SEGMENTS_PORT)->BSRR = code | (~code << 16);
183 (&DIGITS_PORT)->BRR = dig1;
184 currentDisplayDigit = 2;
185 break;
186 }
187 case 2:{
188 uint16_t code = codeOfDigit((number / 10) % 10) | codeOfDots(1);
189 (&SEGMENTS_PORT)->BSRR = code | (~code << 16);
190 (&DIGITS_PORT)->BRR = dig2;

193 }
